const picks = [ { id: 1, name: "Trae Young", matchup: "ATL vs BOS", stat: "Points", line: 27.5 }, { id: 2, name: "Jayson Tatum", matchup: "BOS @ ATL", stat: "Rebounds", line: 8.5 }, { id: 3, name: "Stephen Curry", matchup: "GSW vs LAL", stat: "3PT Made", line: 4.5 }, { id: 4, name: "LeBron James", matchup: "LAL @ GSW", stat: "Assists", line: 7.5 } ]; const picksGrid = document.getElementById("picksGrid"); const slipItems = document.getElementById("slipItems"); const slip = []; function renderPicks() { picksGrid.innerHTML = ""; picks.forEach((pick) => { const card = document.createElement("div"); card.className = "pick-card"; card.innerHTML = `
${pick.name}
${pick.matchup} • ${pick.stat}
${pick.line}
`; picksGrid.appendChild(card); }); document.querySelectorAll(".choice-btn").forEach((btn) => { btn.addEventListener("click", () => { const id = Number(btn.dataset.id); const choice = btn.dataset.choice; addToSlip(id, choice); }); }); } function addToSlip(id, choice) { const existing = slip.find((item) => item.id === id); const pick = picks.find((p) => p.id === id); if (!pick) return; if (existing) { existing.choice = choice; } else { slip.push({ ...pick, choice }); } renderSlip(); } function renderSlip() { slipItems.innerHTML = ""; if (!slip.length) { slipItems.innerHTML = `

No picks selected yet.

`; return; } slip.forEach((item) => { const div = document.createElement("div"); div.className = "slip-item"; div.innerHTML = ` ${item.name}
${item.choice} ${item.line} ${item.stat}
${item.matchup} `; slipItems.appendChild(div); }); } renderPicks(); renderSlip();
